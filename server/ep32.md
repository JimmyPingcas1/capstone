    // ================= DEVICE CONTROL (Server-driven) =================
    // AI sends ON/OFF commands via sensor response - no manual fixing needed

    // ================= HANDLE AI =================
    void handleAI(String response) {
    StaticJsonDocument<512> doc;
    if (deserializeJson(doc, response)) {
        Serial.println("❌ JSON Parse Failed");
        return;
    }

    // Keep aiActive in sync with server payloads from HTTP or WebSocket.
    if (doc.containsKey("ai_mode")) {
        aiActive = doc["ai_mode"].as<bool>();
    } else if (doc.containsKey("aiMode")) {
        aiActive = doc["aiMode"].as<bool>();
    } else if (doc.containsKey("devices_on") || doc.containsKey("devices_off")) {
        aiActive = true;
    }

    Serial.print("⚡ AI Mode: ");
    Serial.println(aiActive ? "ON" : "OFF");

    // Print current device states
    Serial.print("💡 Current Device States — Heater: ");
    Serial.print(heaterOn ? "ON" : "OFF");
    Serial.print(", Aerator: ");
    Serial.print(aeratorOn ? "ON" : "OFF");
    Serial.print(", WaterPump: ");
    Serial.println(pumpOn ? "ON" : "OFF");

    // ✅ AI Server now controls device ON/OFF directly via response
    if (!aiActive) {
        Serial.println("⚠ AI is OFF → skipping device control");
        return;
    }

    Serial.println("\n🤖 AI Device Control Response");

    // Process devices_on using server's boolean keys: heater, waterpump.
    if (doc.containsKey("devices_on")) {
        JsonObject devicesOn = doc["devices_on"];

        if (devicesOn.containsKey("heater")) {
        heaterOn = devicesOn["heater"].as<bool>();
        }
        if (devicesOn.containsKey("waterpump")) {
        pumpOn = devicesOn["waterpump"].as<bool>();
        }
    }

    // Process devices_off boolean flags.
    if (doc.containsKey("devices_off")) {
        JsonObject devicesOff = doc["devices_off"];

        if (devicesOff.containsKey("heater") && devicesOff["heater"].as<bool>()) {
        heaterOn = false;
        }
        if (devicesOff.containsKey("waterpump") && devicesOff["waterpump"].as<bool>()) {
        pumpOn = false;
        }
    }

    // AI controls only heater and waterpump. Aerator remains manual-only.
    digitalWrite(HEATER_PIN, heaterOn ? HIGH : LOW);
    digitalWrite(WATERPUMP_PIN, pumpOn ? HIGH : LOW);

    Serial.print("🤖 AI Applied — Heater=");
    Serial.print(heaterOn ? "ON" : "OFF");
    Serial.print(" | WaterPump=");
    Serial.print(pumpOn ? "ON" : "OFF");
    Serial.print(" | Aerator(manual)=");
    Serial.println(aeratorOn ? "ON" : "OFF");

    if (doc.containsKey("danger") && !doc["danger"].as<bool>()) {
        Serial.println("✅ All parameters normal");
    }

    Serial.println("=====================================");
    }

    // ================= SEND SENSOR =================
    void sendSensor() {
    // Read real sensor values first
    if (!readSensors()) {
        Serial.println("❌ Sensor read failed, skipping send");
        return;
    }
    
    Serial.println("\n📤 Sending sensor data to AI...");
    Serial.print("   Temp: "); Serial.print(temperature); Serial.print("°C | ");
    Serial.print("pH: "); Serial.print(phLevel); Serial.print(" | ");
    Serial.print("Turbidity: "); Serial.print(turbidity); Serial.print(" NTU | ");
    Serial.print("Ammonia: "); Serial.print(ammonia); Serial.println(" ppm");
    
    HTTPClient http;
    http.begin(sensorUrl);
    http.addHeader("Content-Type", "application/json");

    StaticJsonDocument<256> doc;
    doc["temperature"] = temperature;
    doc["turbidity"]   = turbidity;
    doc["ph"]          = phLevel;
    doc["ammonia"]     = ammonia;

    String body;
    serializeJson(doc, body);

    int code = http.POST(body);
    Serial.print("📡 AI Response Code: ");
    Serial.println(code);

    if (code == 200) {
        String response = http.getString();
        Serial.println("📥 AI Response:");
        Serial.println(response);

        handleAI(response);
    } else {
        Serial.println("❌ Failed to send sensor data");
    }

    http.end();
    }

    // No local fixing simulation - AI server controls device state continuously

    // ================= AI WEBSOCKET =================
    void webSocketEvent(WStype_t type, uint8_t * payload, size_t length) {
    switch(type) {
        case WStype_DISCONNECTED:
        Serial.println("❌ AI WebSocket Disconnected — will auto-reconnect");
        break;
        case WStype_CONNECTED:
        Serial.println("✅ AI WebSocket Connected");
        webSocket.sendTXT("{\"request_status\": true}");
        break;
        case WStype_TEXT:
        Serial.println("📩 AI WebSocket Message:");
        Serial.println((char*)payload);
        handleAI(String((char*)payload));
        break;
        default: break;
    }
    }


    // ================= BUTTON (MANUAL) WEBSOCKET =================
    void buttonWebSocketEvent(WStype_t type, uint8_t * payload, size_t length) {
        switch(type) {

            case WStype_DISCONNECTED:
                Serial.println("❌ Manual WebSocket Disconnected — will auto-reconnect");
                handshakeSent = false;  // Reset flag on disconnect
                statesSynced = false;   // Reset sync flag
                break;

            case WStype_CONNECTED:
                Serial.println("✅ Manual WebSocket Connected");
                
                // Small delay to ensure connection is fully established
                delay(150);
                
                // 1️⃣ Send handshake request to server automatically
                Serial.println("📤 Sending handshake to server...");
                buttonWebSocket.sendTXT("{\"type\": \"handshake\"}");
                
                // 2️⃣ Immediately request current device states
                delay(200);
                Serial.println("📥 Requesting device states from server...");
                if (buttonWebSocket.sendTXT("{\"type\": \"state_request\"}")) {
                    Serial.println("✅ State request sent");
                    handshakeSent = true;
                    lastHandshakeAttempt = millis();
                } else {
                    Serial.println("⚠️  State request failed");
                    handshakeSent = false;
                }
                break;

            case WStype_TEXT: {
                Serial.println("📩 Manual WebSocket Message Received:");
                Serial.println((char*)payload);

                StaticJsonDocument<512> doc;
                DeserializationError error = deserializeJson(doc, payload);

                if (!error) {
                    // Check if this is a state response
                    bool hasDevices = doc.containsKey("devices");
                    
                    if (hasDevices) {
                        bool previousAiActive = aiActive;
                        
                        // Update AI mode first
                        if (doc.containsKey("aiMode")) aiActive = doc["aiMode"].as<bool>();

                    JsonObject devices = doc["devices"];

                    // Aerator is manual-only, so always read/apply it from manual state.
                    if (devices.containsKey("aerator")) {
                        aeratorOn = devices["aerator"].as<bool>();
                    }

                    // If AI just turned ON, hand over only heater/pump to AI.
                    if (aiActive && !previousAiActive) {
                        heaterOn = false;
                        pumpOn = false;

                        digitalWrite(HEATER_PIN, LOW);
                        digitalWrite(WATERPUMP_PIN, LOW);

                        Serial.println("🤖 AI Mode activated - Heater/WaterPump now AI-controlled; Aerator stays manual");
                    }

                    // Parse heater/waterpump from manual state only when AI is OFF.
                    if (!aiActive) {
                        if (devices.containsKey("heater"))    heaterOn  = devices["heater"].as<bool>();
                        if (devices.containsKey("waterpump")) pumpOn    = devices["waterpump"].as<bool>();
                    }

                        Serial.print("💡 States: Heater=");
                        Serial.print(heaterOn ? "ON" : "OFF");
                        Serial.print(" | Aerator=");
                        Serial.print(aeratorOn ? "ON" : "OFF");
                        Serial.print(" | Pump=");
                        Serial.println(pumpOn ? "ON" : "OFF");

                        // 3️⃣ Apply states to GPIO pins
                        if (!aiActive) {
                        // Manual mode: apply all device states.
                            digitalWrite(HEATER_PIN, heaterOn ? HIGH : LOW);
                            digitalWrite(AERATOR_PIN, aeratorOn ? HIGH : LOW);
                            digitalWrite(WATERPUMP_PIN, pumpOn ? HIGH : LOW);
                            
                            Serial.println("✅ Manual states applied to GPIO pins");
                            handshakeSent = true;
                            statesSynced = true;  // Mark states as successfully synced
                        } else {
                        // AI mode: keep heater/waterpump under AI; still apply aerator (manual-only).
                        digitalWrite(AERATOR_PIN, aeratorOn ? HIGH : LOW);
                        Serial.println("⚠️  AI Mode ON → Heater/WaterPump by AI, Aerator from manual state");
                            handshakeSent = true;
                            statesSynced = true;
                        }
                    }

                } else {
                    Serial.print("❌ JSON Parse Error: ");
                    Serial.println(error.c_str());
                }
                break;
            }

            default:
                break;
        }
    }

    void loop() {
    webSocket.loop();        // AI WS
    buttonWebSocket.loop();  // Manual WS

    // 🔄 Retry handshake & request states if not connected properly
    if (!handshakeSent && (millis() - lastHandshakeAttempt > handshakeRetryInterval)) {
        if (buttonWebSocket.isConnected()) {
            Serial.println("🔄 Retrying handshake & state request...");
            buttonWebSocket.sendTXT("{\"type\": \"handshake\"}");
            delay(150);
            buttonWebSocket.sendTXT("{\"type\": \"state_request\"}");
            Serial.println("📤 Handshake & state request resent");
            lastHandshakeAttempt = millis();
        }
    }

    // 📡 Periodically request device states from server
    if (buttonWebSocket.isConnected() && (millis() - lastStateRequest > stateRequestInterval)) {
        Serial.println("📡 Requesting states from server...");
        if (buttonWebSocket.sendTXT("{\"type\": \"state_request\"}")) {
            Serial.println("✅ State request sent");
        }
        lastStateRequest = millis();
    }

    // 📤 Send sensor data periodically (AI responds with ON/OFF commands)
    if (millis() - lastSend > sensorInterval) {
        lastSend = millis();
        sendSensor();
    }

    delay(200);
    }
